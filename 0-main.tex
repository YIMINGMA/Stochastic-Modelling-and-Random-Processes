\NewDocumentCommand{\prob}{ s O{} > {\SplitArgument{1}{|}}m }
{
    \mbb{P}
    \IfBooleanTF{#1}
    { \PPauto #3 }
    { \PPfixed {#2} #3 }
}

{
    \mbb{E}

    { \PPauto #3 }
    { \PPfixed {#2} #3 }
 }

\NewDocumentCommand{\Var}{ s O{} > {\SplitArgument{1}{|}}m }
{

    { \PPfixed {#2} #3 }
}

\NewDocumentCommand{\Cov}{ s O{} > {\SplitArgument{1}{|}}m }

}

\NewDocumentCommand{\PPauto}{mm}
{
   \left(
   \IfNoValueTF{#2} { #1 } { #1 \;\middle|\; #2 }
   \right)
}

\NewDocumentCommand{\PPfixed}{mmm}
{
   \mathopen{#1(}
   \IfNoValueTF{#3} { #2 } { #2 \mathrel{#1|} #3 }
   \mathclose{#1)}
}
